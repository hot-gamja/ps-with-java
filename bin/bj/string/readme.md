```java
length()              // 길이
charAt(i)             // i번째 문자
substring(a)          // a부터 끝까지
substring(a, b)       // a부터 b-1까지
contains("x")         // 포함 여부
indexOf("x")          // 처음 위치
lastIndexOf("x")      // 마지막 위치
equals("x")           // 내용 비교
toUpperCase()         // 대문자
toLowerCase()         // 소문자
trim()                // 앞뒤 공백 제거
replace("a", "b")     // 치환
split(",")            // 나누기
startsWith("ab")      // 시작 확인
endsWith("yz")        // 끝 확인
isEmpty()             // 빈 문자열 확인
isBlank()             // 공백 포함 빈 문자열 확인
```
